```ruby

func ascending(currentElement: Int, nextElement: Int) -> Bool {
    return currentElement > nextElement
}
func descending(currentElement: Int, nextElement: Int) -> Bool {
    return currentElement < nextElement
}
// 함수를 리턴값으로 사용하는 함수
func createComparator(isAscending: Bool = true) -> (Int, Int) -> Bool {
    return isAscending ? ascending : descending
}
// 함수를 매개변수로 사용하는 함수
func bubbleSort(
    _ arr: inout Array<Int>,
    comparator: (Int, Int) -> Bool = createComparator()
) {
    var tmp: Int
    for i in (1..<arr.count).reversed() {
        for j in 0..<i {
            if comparator(arr[j], arr[j + 1]) {
                tmp = arr[j]
                arr[j] = arr[j + 1]
                arr[j + 1] = tmp
            }    
        }
    }
}
func printArr(arr: Array<Int>) {
    for i in arr {
        print(i, terminator: " ")
    }
    print()
}
var arr = [1, 2, 4, 8, 3, 5, 6, 7, 9]
bubbleSort(&arr)
printArr(arr: arr) // 1 2 3 4 5 6 7 8 9
// 함수를 변수에 저장
var myComparator = createComparator(isAscending: false) // descending
bubbleSort(&arr, comparator: myComparator)
printArr(arr: arr) // 9 8 7 6 5 4 3 2 1
```
